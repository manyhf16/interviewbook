# 用递归实现N！ = N*(N-1)*(N-2)*......*2*1。

# 参考答案

```
public class Test185{
	public static void main(String[] args) {
		int n = 5;
		System.out.print(n+"!=");
		factorial(5);
	}
	public static long factorial(int n){
		if(n==1){
			System.out.println(1);
			return 1;
		}
		System.out.print(n+"*");
		return n*factorial(n-1);
	}
}
```
